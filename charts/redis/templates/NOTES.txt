** Retrieve the password **

  kubectl get secret --namespace {{ include "redis.namespace" . }} {{ include "redis.secretName" . }} \

** Connect with redis-cli **

  export REDIS_PASSWORD=$(kubectl get secret --namespace {{ include "redis.namespace" . }} {{ include "redis.secretName" . }} \
    -o jsonpath="{.data.{{ .Values.auth.existingSecretPasswordKey }}}" | base64 -d)

  kubectl run --namespace {{ include "redis.namespace" . }} redis-client --rm --tty -i --restart='Never' \
    --image {{ include "redis.image" . }} \
    --env REDIS_PASSWORD=$REDIS_PASSWORD \
    -- redis-cli -h {{ include "redis.master.fullname" . }} -a $REDIS_PASSWORD
